140void PMNS_Avg::BuildKE(double L)
141{
142 double lenghtEV = L * kKm2eV; // L in eV-1
143 double bufK = lenghtEV * 0.5; // L/2 in eV-1
144
145 complexD buffer[3];
146
147 for (int i = 0; i < fNumNus; i++) {
148 complexD Hms_kl;
149
150 for (int l = 0; l < fNumNus; l++) {
151 buffer[l] = 0;
152
153 for (int k = 0; k < fNumNus; k++) {
154 if (k <= l)
155 Hms_kl = fHms[k][l];
156 else
157 Hms_kl = conj(fHms[l][k]);
158
159 if (fIsNuBar && k != l) Hms_kl = conj(Hms_kl);
160
161 buffer[l] += conj(fEvec[k][i]) * Hms_kl;
162 }
163 }
164
165 for (int j = 0; j <= i; j++) {
166 fKmass[i][j] = 0;
167
168 for (int l = 0; l < fNumNus; l++) {
169 fKmass[i][j] += buffer[l] * fEvec[l][j];
170 }
171
172 complexD C;
173
174 if (i == j) { C = complexD(1, 0); }
175 else {
176 double arg = (fEval[i] - fEval[j]) * lenghtEV;
177
178 C = (complexD(cos(arg), sin(arg)) - complexD(1, 0.0)) /
179 complexD(0.0, arg);
180 }
181
182 fKmass[i][j] *= bufK * C;
183
184 if (i != j) fKmass[j][i] = conj(fKmass[i][j]);
185 }
186 }
187}

198void PMNS_Avg::BuildKcosT(double L)
199{
200 UpdateHam();
201
202 double dL = LnDerivative() * kKm2eV;
203

206 fKflavor[i][j] = dL * fHam[i][j];

213//.............................................................................
217double PMNS_Avg::LnDerivative()
218{
219 double dL = 0;
220
221 double L1 = pow(fPrem.GetPremLayers()[fLayer].radius, 2) - fminRsq;
222
223 double L2 = -fminRsq;
224 if (fLayer > 0) L2 += pow(fPrem.GetPremLayers()[fLayer - 1].radius, 2);
225
226 bool ismin = (L2 <= 0 && fcosT < 0);
227
228 if (ismin)
229 dL = 2 * pow(fDetRadius, 2) * fcosT * pow(L1, -0.5);
230 else
231 dL = pow(fDetRadius, 2) * fcosT * (pow(L1, -0.5) - pow(L2, -0.5));
232
233 if (ismin) fdl = 1;
234
235 fLayer += fdl;
236
237 return dL;
238}

366//.............................................................................
371void PMNS_Avg::PropagatePathTaylor(NuPath p)
372{
373 // Set the neutrino path
374 SetCurPath(p);
375
376 // Solve for eigensystem
377 SolveHam();
378
379 // Get the evolution matrix in mass basis
380 double LengthIneV = kKm2eV * p.length;
381 for (int i = 0; i < fNumNus; i++) {
382 double arg = fEval[i] * LengthIneV;
383 fPhases[i] = complexD(cos(arg), -sin(arg));
384 }
385
386 // Rotate S in flavor basis
387 rotateS();
388
389 // if avg on E
390 if (fdInvE != 0) {
391 // Build KE in mass basis
392 BuildKE(p.length);
393
394 // Rotate KE in flavor basis
395 rotateK();
396
397 // Multiply this layer K's with the previous path K's
398 MultiplicationRuleK(fKInvE);
399 }
400
401 // if avg on cosT
402 if (fdcosT != 0) {
403 // Build KcosT in mass basis
404 BuildKcosT(p.length);
405
406 // Multiply this layer K's with the previous path K's
407 MultiplicationRuleK(fKcosT);
408 }
409
410 // Multiply this layer S's with the previous path S's
411 MultiplicationRuleS();
412}

426void PMNS_Avg::SolveK(complexD K[3][3], vectorD& lambda, matrixC& V)
427{
428 double fEvalGLoBES[3];
429 complexD fEvecGLoBES[3][3];
430
431 // Solve K for eigensystem using the GLoBES method
432 zheevh3(K, fEvecGLoBES, fEvalGLoBES);
433
434 // Fill flambdaInvE and fVInvE vectors from GLoBES arrays
435 for (int i = 0; i < fNumNus; i++) {
436 lambda[i] = fEvalGLoBES[i];
437 for (int j = 0; j < fNumNus; j++) { V[i][j] = fEvecGLoBES[i][j]; }
438 }

460double PMNS_Avg::AvgFormula(int flvi, int flvf, double dbin, vectorD lambda,
461 matrixC V)
462{
463 vectorC SV = vectorC(fNumNus, 0);
464
465 for (int i = 0; i < fNumNus; i++) {
466 for (int k = 0; k < fNumNus; k++) {
467 SV[i] += fevolutionMatrixS[flvf][k] * V[k][i];
468 }
469 }
470
471 complexD buffer[3];
472
473 for (int n = 0; n < fNumNus; n++) { buffer[n] = SV[n] * conj(V[flvi][n]); }
474
475 complexD sinc[fNumNus][fNumNus];
476
477 for (int j = 0; j < fNumNus; j++) {
478 sinc[j][j] = 1;
479
480 for (int i = 0; i < j; i++) {
481 double arg = (lambda[i] - lambda[j]) * dbin / 2;
482 sinc[i][j] = sin(arg) / arg;
483 sinc[j][i] = sinc[i][j];
484 }
485 }
486
487 complexD P = 0;
488
489 for (int j = 0; j < fNumNus; j++) {
490 for (int i = 0; i < fNumNus; i++) {
491 P += buffer[i] * conj(buffer[j]) * sinc[j][i];
492 }
493 }
494
495 return real(P);
496}

558double PMNS_Avg::AvgProbLoE(int flvi, int flvf, double LoE, double dLoE)
559{
560 if (LoE <= 0) return 0;
561
562 if (fNuPaths.empty()) return 0;
563
564 // Don't average zero width
565 if (dLoE <= 0) return Prob(flvi, flvf, fPath.length / LoE);
566
567 // Get sample points for this bin
568 vectorD samples = GetSamplePointsAvgClass(LoE, dLoE);
569
570 double avgprob = 0;
571 double L = fPath.length;
572 double sumw = 0;
573
574 // Loop over all sample points
575 for (int j = 1; j < int(samples.size()); j++) {
576 // Set (L/E)^-2 weights
577 double w = 1. / pow(samples[j], 2);
578
579 avgprob += w * AvgAlgo(flvi, flvf, samples[j], samples[0], L);
580
581 // Increment sum of weights
582 sumw += w;
583 }
584
585 // Return weighted average of probabilities
586 return avgprob / sumw;
587}

601double PMNS_Avg::AvgAlgo(int flvi, int flvf, double LoE, double dLoE, double L)
602{
603 // Set width bin as 1/E
604 double d1oE = dLoE / L;
605
606 // reset K et S et Ve et lambdaE
607 InitializeTaylorsVectors();
608
609 SetEnergy(L / LoE);
610 SetwidthBin(d1oE, 0);
611
612 // Try caching if activated
613 if (!TryCacheK()) {
614
615 // Propagate -> get S and K matrix (on the whole path)
616 PropagateTaylor();
617
618 // DiagolK -> get VE and lambdaE
619 SolveK(fKInvE, flambdaInvE, fVInvE);
620
621 // Fill cache if activated
622 FillCacheK();
623
624 }
625
626 // return fct avr proba
627 return AvgFormula(flvi, flvf, d1oE / kGeV2eV, flambdaInvE, fVInvE);
628}

959 imag(samples[k][l]), imag(samples[0][0]));

961 // Increment sum of weights
962 sumw += w;
963 }
964 }
965
966 // Return average of probabilities
967 return avgprob / ((sumw));

1256 for (int j = 0; j < fNumNus; j++) {
1257 for (int i = 0; i < fNumNus; i++) {

1264
1265 for (int j = 0; j < fNumNus; j++) {
1266 for (int i = 0; i < fNumNus; i++) {
1267 P += buffer[i] * conj(buffer[j]) * expo[j][i];
