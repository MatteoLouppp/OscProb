124void PMNS_TaylorExp::BuildKE(double L)
125{
126 double lenghtEV = L * kKm2eV ; // L in eV-1
127 double bufK = lenghtEV * 0.5 ; // L/2 in eV-1
128
129 complexD buffer[3];
130
131 for(int i = 0 ; i<fNumNus ; i++){
132
133 complexD Hms_kl;
134
135 for(int l = 0 ; l<fNumNus ; l++){
136
137 buffer[l] = 0;
138
139 for(int k = 0 ; k<fNumNus ; k++){
140
141 if (k<=l)
142 Hms_kl = fHms[k][l];
143 else
144 Hms_kl = conj(fHms[l][k]);
145
146 if(fIsNuBar && k!=l)
147 Hms_kl = conj(Hms_kl);
148
149 buffer[l] += conj(fEvec[k][i]) * Hms_kl;
150 }
151 }
152
153 for(int j = 0 ; j<=i ; j++){
154
155 fKmass[i][j] = 0;
156
157 for (int l = 0 ; l<fNumNus ; l++){
158 fKmass[i][j] += buffer[l] * fEvec[l][j] ;
159 }
160
161 complexD C;
162
163 if(i == j){
164 C = complexD(1,0);
165 }
166 else {
167 double arg = (fEval[i] - fEval[j]) * lenghtEV ;
168
169 C = ( complexD(cos(arg) , sin(arg)) - complexD(1 , 0.0) ) / complexD(0.0 , arg) ;
170
171 }
172
173 fKmass[i][j] *= bufK * C ;
174
175 if(i != j)
176 fKmass[j][i] = conj(fKmass[i][j]);
177
178 }
179 }
180
181}

192void PMNS_TaylorExp::BuildKcosT(double L)
193{
194 double lv = 2 * kGeV2eV * fEnergy; // 2E in eV
195 double kr2GNe = kK2 * M_SQRT2 * kGf;
196 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
197
198 matrixC Hbar = matrixC(fNumNus, vectorC(fNumNus, 0));
199
200 for (int i = 0; i < fNumNus; i++) {
201 Hbar[i][i] = fHms[i][i] / lv;
202 for (int j = i + 1; j < fNumNus; j++) {
203 if (!fIsNuBar)
204 Hbar[i][j] = fHms[i][j] / lv;
205 else
206 Hbar[i][j] = conj(fHms[i][j]) / lv;
207 }
208 }
209 if (!fIsNuBar)
210 Hbar[0][0] += kr2GNe;
211 else
212 Hbar[0][0] -= kr2GNe;
213
214 double dL = LnDerivative() * kKm2eV;
215
216 for(int j = 0 ; j<fNumNus ; j++){
217 for(int i = 0 ; i<=j ; i++){
218 fKflavor[i][j] = dL * Hbar[i][j];
219
220 if(i != j){
221 fKflavor[j][i] = conj(fKflavor[i][j]);
222 }
223 }
224 }
225
226}

228//.............................................................................
232double PMNS_TaylorExp::LnDerivative()
233{
234 double dL = 0;
235
236 double L1 = pow(fPremLayers[flayer].radius, 2) - fminRsq;
237
238 double L2 = -fminRsq;
239 if (flayer > 0) L2 += pow(fPremLayers[flayer - 1].radius, 2);
240
241 bool ismin = (L2 <= 0 && fcosT < 0);
242
243 if (ismin)
244 dL = 2 * pow(fDetRadius,2) * fcosT * pow( L1 , -0.5) ;
245 else
246 dL = pow(fDetRadius,2) * fcosT * ( pow( L1 , -0.5) - pow( L2 , -0.5));
247
248 if (ismin) fdl = 1;
249
250 flayer += fdl;
251
252 return dL;
253}

389//.............................................................................
394void PMNS_TaylorExp::PropagatePathTaylor(NuPath p)
395{
396
397 // Set the neutrino path
398 SetCurPath(p);
399
400 // Solve for eigensystem
401 SolveHam();
402
403 // Get the evolution matrix in mass basis
404 double LengthIneV = kKm2eV * p.length;
405 for (int i = 0; i < fNumNus; i++) {
406 double arg = fEval[i] * LengthIneV;
407 fPhases[i] = complexD(cos(arg), -sin(arg));
408 }
409
410 // Rotate S in flavor basis
411 rotateS();
412
413 // if avg on E
414 if(fdInvE != 0){
415
416 // Build KE in mass basis
417 BuildKE(p.length);
418
419 // Rotate KE in flavor basis
420 rotateK();
421
422 // Multiply this layer K's with the previous path K's
423 MultiplicationRuleK(fKInvE);
424
425 }
426
427 // if avg on cosT
428 if(fdcosT != 0){
429
430 // Build KcosT in mass basis
431 BuildKcosT(p.length);
432
433 // Multiply this layer K's with the previous path K's
434 MultiplicationRuleK(fKcosT);
435
436 }
437
438 // Multiply this layer S's with the previous path S's
439 MultiplicationRuleS();
440
441}

489double PMNS_TaylorExp::AvgFormula(int flvi, int flvf, double dbin, vectorD lambda, matrixC V)
490{
491
492 vectorC SV = vectorC(fNumNus, 0);
493
494 for(int i = 0 ; i<fNumNus ; i++) {
495 for(int k = 0 ; k<fNumNus ; k++){
496 SV[i] += fevolutionMatrixS[flvf][k] *V[k][i];
497 }
498 }
499
500 complexD buffer [3];
501
502 for( int n = 0 ; n<fNumNus ; n++){
503 buffer[n] = SV[n] * conj(V[flvi][n]);
504 }
505
506 complexD sinc[fNumNus][fNumNus];
507
508 for(int j = 0 ; j<fNumNus ; j++){
509
510 sinc[j][j] = 1;
511
512 for(int i = 0 ; i<j ; i++){
513 double arg = (lambda[i] - lambda[j]) * dbin / 2;
514 sinc[i][j] = sin(arg) / arg;
515 sinc[j][i] = sinc[i][j];
516 }
517 }
518
519 complexD P = 0;
520
521 for(int j = 0 ; j<fNumNus ; j++){
522 for(int i = 0 ;i<fNumNus ;i++){
523 P += buffer[i] *conj(buffer[j]) * sinc[j][i];
524 }
525 }
526
527 return real(P);
528}

551double PMNS_TaylorExp::AvgProb(int flvi, int flvf, double E , double dE)
552{
553 if (E <= 0) return 0;
554
555 if (fNuPaths.empty()) return 0;
556
557 // Don't average zero width
558 if (dE <= 0) return Prob(flvi, flvf, E);
559
560 vectorD Ebin = ConvertEtoLoE(E,dE);
561
562 //return fct avr proba
563 return AvgProbLoE(flvi, flvf, Ebin[0], Ebin[1]);
564}

584double PMNS_TaylorExp::AvgProbLoE(int flvi, int flvf, double LoE , double dLoE)
585{
586 if (LoE <= 0) return 0;
587
588 if (fNuPaths.empty()) return 0;
589
590 // Don't average zero width
591 if (dLoE <= 0) return Prob(flvi, flvf, fPath.length / LoE);
592
593 // Get sample points for this bin
594 vectorD samples = GetSamplePoints(LoE, dLoE);
595
596 double avgprob = 0;
597 double L = fPath.length;
598 double sumw = 0;
599
600 // Loop over all sample points
601 for (int j = 1; j < int(samples.size()); j++) {
602
603 double w = 1. / pow(samples[j], 2);
604
605 avgprob += w * AvgAlgo(flvi, flvf, samples[j], samples[0],L);
606
607 sumw += w;
608
609 }
610
611 // Return average of probabilities
612 return avgprob / sumw;
613}

619double PMNS_TaylorExp::AvgAlgo(int flvi, int flvf, double LoE , double dLoE, double L)
620{
621 // Set width bin as 1/E
622 double d1oE = dLoE / L;
623
624 // reset K et S et Ve et lambdaE
625 InitializeTaylorsVectors();
626
627 SetEnergy(L / LoE);
628 SetwidthBin(d1oE, 0);
629
630 //Propagate -> get S and K matrix (on the whole path)
631 PropagateTaylor();
632
633 //DiagolK -> get VE and lambdaE
634 SolveK(fKInvE,flambdaInvE,fVInvE);
635
636 //return fct avr proba
637 return AvgFormula(flvi, flvf, d1oE/ kGeV2eV , flambdaInvE, fVInvE);

1001void PMNS_TaylorExp::RotateDensityM(bool to_mass, matrixC V)
1002{
1003 matrixC Buffer = matrixC(fNumNus, vectorC(fNumNus, 0));
1004
1005 // buffer = rho . U
1006 for (int i = 0; i < fNumNus; i++) {
1007 for (int j = 0; j < fNumNus; j++) {
1008 for (int k = 0; k < fNumNus; k++) {
1009 if (to_mass)
1010 Buffer[i][j] += fdensityMatrix[i][k] * V[k][j];
1011 else
1012 Buffer[i][j] += fdensityMatrix[i][k] * conj(V[j][k]);
1013 }
1014 }
1015 }
1016
1017 // rho = U^\dagger . buffer = U^\dagger . rho . U
1018 // Final matrix is Hermitian, so copy upper to lower triangle
1019 for (int i = 0; i < fNumNus; i++) {
1020 for (int j = i; j < fNumNus; j++) {
1021 fdensityMatrix[i][j] = 0;
1022 for (int k = 0; k < fNumNus; k++) {
1023 if (to_mass)
1024 fdensityMatrix[i][j] += conj(V[k][i]) * Buffer[k][j];
1025 else
1026 fdensityMatrix[i][j] += V[i][k] * Buffer[k][j];
1027 }
1028 if (j > i) fdensityMatrix[j][i] = conj(fdensityMatrix[i][j]);
1029 }
1030 }
1031
1032}

1038void PMNS_TaylorExp::HadamardProduct(vectorD lambda, double dbin)
1039{
1040 matrixC sinc = matrixC(fNumNus, vectorC(fNumNus, 0));
1041 for(int j=0 ; j<fNumNus ; j++){
1042 for(int i = 0 ; i<j ; i++){
1043 double arg = (lambda[i] - lambda[j]) * dbin ;
1044 sinc[i][j] = sin(arg)/arg;
1045
1046 sinc[j][i] = sinc[i][j];
1047 }
1048 }
1049
1050 for(int i=0 ; i<fNumNus ; i++){
1051 sinc[i][i] = 1;
1052 }
1053
1054 for(int j=0 ; j<fNumNus ; j++){
1055 for(int i = 0 ; i<fNumNus ; i++){
1056 fdensityMatrix[i][j] = fdensityMatrix[i][j] * sinc[i][j];
1057 }
1058 }
