127void PMNS_TaylorExp::BuildKE(double L)
128{
129 double lenghtEV = L * kKm2eV; // L in eV-1
130 double bufK = lenghtEV * 0.5; // L/2 in eV-1
131
132 complexD buffer[3];
133
134 for (int i = 0; i < fNumNus; i++) {
135 complexD Hms_kl;
136
137 for (int l = 0; l < fNumNus; l++) {
138 buffer[l] = 0;
139
140 for (int k = 0; k < fNumNus; k++) {
141 if (k <= l)
142 Hms_kl = fHms[k][l];
143 else
144 Hms_kl = conj(fHms[l][k]);
145
146 if (fIsNuBar && k != l) Hms_kl = conj(Hms_kl);
147
148 buffer[l] += conj(fEvec[k][i]) * Hms_kl;
149 }
150 }
151
152 for (int j = 0; j <= i; j++) {
153 fKmass[i][j] = 0;
154
155 for (int l = 0; l < fNumNus; l++) {
156 fKmass[i][j] += buffer[l] * fEvec[l][j];
157 }
158
159 complexD C;
160
161 if (i == j) { C = complexD(1, 0); }
162 else {
163 double arg = (fEval[i] - fEval[j]) * lenghtEV;
164
165 C = (complexD(cos(arg), sin(arg)) - complexD(1, 0.0)) /
166 complexD(0.0, arg);
167 }
168
169 fKmass[i][j] *= bufK * C;
170
171 if (i != j) fKmass[j][i] = conj(fKmass[i][j]);
172 }
173 }
174}

224double PMNS_TaylorExp::LnDerivative()
225{
226 double dL = 0;
227
228 double L1 = pow(fPremLayers[flayer].radius, 2) - fminRsq;
229
230 double L2 = -fminRsq;
231 if (flayer > 0) L2 += pow(fPremLayers[flayer - 1].radius, 2);
232
233 bool ismin = (L2 <= 0 && fcosT < 0);
234
235 if (ismin)
236 dL = 2 * pow(fDetRadius, 2) * fcosT * pow(L1, -0.5);
237 else
238 dL = pow(fDetRadius, 2) * fcosT * (pow(L1, -0.5) - pow(L2, -0.5));
239
240 if (ismin) fdl = 1;
241
242 flayer += fdl;
243
244 return dL;
245}

373//.............................................................................
378void PMNS_TaylorExp::PropagatePathTaylor(NuPath p)
379{
380 // Set the neutrino path
381 SetCurPath(p);
382
383 // Solve for eigensystem
384 SolveHam();
385
386 // Get the evolution matrix in mass basis
387 double LengthIneV = kKm2eV * p.length;
388 for (int i = 0; i < fNumNus; i++) {
389 double arg = fEval[i] * LengthIneV;
390 fPhases[i] = complexD(cos(arg), -sin(arg));
391 }
392
393 // Rotate S in flavor basis
394 rotateS();
395
396 // if avg on E
397 if (fdInvE != 0) {
398 // Build KE in mass basis
399 BuildKE(p.length);
400
401 // Rotate KE in flavor basis
402 rotateK();
403
404 // Multiply this layer K's with the previous path K's
405 MultiplicationRuleK(fKInvE);
406 }
407
408 // if avg on cosT
409 if (fdcosT != 0) {
410 // Build KcosT in mass basis
411 BuildKcosT(p.length);
412
413 // Multiply this layer K's with the previous path K's
414 MultiplicationRuleK(fKcosT);
415 }
416
417 // Multiply this layer S's with the previous path S's
418 MultiplicationRuleS();
419}

434{

467double PMNS_TaylorExp::AvgFormula(int flvi, int flvf, double dbin,
468 vectorD lambda, matrixC V)
469{
470 vectorC SV = vectorC(fNumNus, 0);
471
472 for (int i = 0; i < fNumNus; i++) {
473 for (int k = 0; k < fNumNus; k++) {
474 SV[i] += fevolutionMatrixS[flvf][k] * V[k][i];
475 }
476 }
477
478 complexD buffer[3];
479
480 for (int n = 0; n < fNumNus; n++) { buffer[n] = SV[n] * conj(V[flvi][n]); }
481
482 complexD sinc[fNumNus][fNumNus];
483
484 for (int j = 0; j < fNumNus; j++) {
485 sinc[j][j] = 1;
486
487 for (int i = 0; i < j; i++) {
488 double arg = (lambda[i] - lambda[j]) * dbin / 2;
489 sinc[i][j] = sin(arg) / arg;
490 sinc[j][i] = sinc[i][j];
491 }
492 }
493
494 complexD P = 0;
495
496 for (int j = 0; j < fNumNus; j++) {
497 for (int i = 0; i < fNumNus; i++) {
498 P += buffer[i] * conj(buffer[j]) * sinc[j][i];
499 }
500 }
501
502 return real(P);
503}

565{
566 if (LoE <= 0) return 0;
567
568 if (fNuPaths.empty()) return 0;

572
573 // Get sample points for this bin
574 vectorD samples = GetSamplePoints(LoE, dLoE);
575
576 double avgprob = 0;
577 double L = fPath.length;
578 double sumw = 0;
579
580 // Loop over all sample points
581 for (int j = 1; j < int(samples.size()); j++) {
582 double w = 1. / pow(samples[j], 2);
583
584 avgprob += w * AvgAlgo(flvi, flvf, samples[j], samples[0], L);
585
586 sumw += w;
587 }
588
589 // Return average of probabilities
590 return avgprob / sumw;

605double PMNS_TaylorExp::AvgAlgo(int flvi, int flvf, double LoE, double dLoE,
606 double L)
607{
608 // Set width bin as 1/E
609 double d1oE = dLoE / L;
610
611 // reset K et S et Ve et lambdaE
612 InitializeTaylorsVectors();
613
614 SetEnergy(L / LoE);
615 SetwidthBin(d1oE, 0);
616
617 // Propagate -> get S and K matrix (on the whole path)
618 PropagateTaylor();
619
620 // DiagolK -> get VE and lambdaE
621 SolveK(fKInvE, flambdaInvE, fVInvE);
622
623 // return fct avr proba
624 return AvgFormula(flvi, flvf, d1oE / kGeV2eV, flambdaInvE, fVInvE);

788double PMNS_TaylorExp::AvgProbLoE(int flvi, int flvf, double LoE, double dLoE,
789 double cosT, double dcosT)
790{
791 if (LoE <= 0) return 0;
792 if (cosT > 0) return 0;
793
794 if (fNuPaths.empty()) return 0;
795
796 // Don't average zero width
797 if (dLoE <= 0 && dcosT == 0) return Prob(flvi, flvf, fPath.length / LoE);

807 int cols = samples[0].size();

811
812 // Loop over all sample points
813 for (int k = 1; k < int(rows); k++) {
814 for (int l = 1; l < int(cols); l++) {
815 double w = 1. / pow(real(samples[k][l]), 2);
816
817 avgprob +=
818 w * AvgAlgo(flvi, flvf, real(samples[k][l]), real(samples[0][0]),
819 imag(samples[k][l]), imag(samples[0][0]));
820
821 sumw += w;
822 }
823 }
824
825 // Return average of probabilities
826 return avgprob / ((sumw));

879double PMNS_TaylorExp::AlgorithmDensityMatrix(int flvi, int flvf)
880{
881 fdensityMatrix[flvi][flvi] = 1;
882
883 RotateDensityM(true, fVcosT);
884 HadamardProduct(flambdaCosT, fdcosT);
885 RotateDensityM(false, fVcosT);
886
887 RotateDensityM(true, fVInvE);
888 HadamardProduct(flambdaInvE, fdInvE / kGeV2eV);
889 RotateDensityM(false, fVInvE);
890
891 RotateDensityM(false, fevolutionMatrixS);
892
893 return real(fdensityMatrix[flvf][flvf]);
894}

904void PMNS_TaylorExp::RotateDensityM(bool to_basis, matrixC V)
905{
906 matrixC Buffer = matrixC(fNumNus, vectorC(fNumNus, 0));
907
908 for (int i = 0; i < fNumNus; i++) {

910 for (int k = 0; k < fNumNus; k++) {
911 if (to_basis)
912 Buffer[i][j] += fdensityMatrix[i][k] * V[k][j];
913 else
914 Buffer[i][j] += fdensityMatrix[i][k] * conj(V[j][k]);
915 }
916 }
917 }
918
919 for (int i = 0; i < fNumNus; i++) {
920 for (int j = i; j < fNumNus; j++) {
921 fdensityMatrix[i][j] = 0;
922 for (int k = 0; k < fNumNus; k++) {
923 if (to_basis)
924 fdensityMatrix[i][j] += conj(V[k][i]) * Buffer[k][j];
925 else
926 fdensityMatrix[i][j] += V[i][k] * Buffer[k][j];
927 }
928 if (j > i) fdensityMatrix[j][i] = conj(fdensityMatrix[i][j]);
929 }
930 }

940void PMNS_TaylorExp::HadamardProduct(vectorD lambda, double dbin)
941{
942 matrixC sinc = matrixC(fNumNus, vectorC(fNumNus, 0));
943 for (int j = 0; j < fNumNus; j++) {
944 for (int i = 0; i < j; i++) {
945 double arg = (lambda[i] - lambda[j]) * dbin;
946 sinc[i][j] = sin(arg) / arg;
947
948 sinc[j][i] = sinc[i][j];
949 }
950 }
951
952 for (int i = 0; i < fNumNus; i++) { sinc[i][i] = 1; }
953
954 for (int j = 0; j < fNumNus; j++) {
955 for (int i = 0; i < fNumNus; i++) {
956 fdensityMatrix[i][j] = fdensityMatrix[i][j] * sinc[i][j];
957 }
958 }

972{
973 // Set a number of sub-divisions to achieve "good" accuracy
974 // This needs to be studied better
975 int n_div = ceil(3 * pow(dLoE / LoE, 0.8) * pow(LoE, 0.3) /
976 sqrt(fAvgProbPrec / 1e-4));

978 //int n_div = 1;

980 // A vector to store sample points
981 vectorD Samples;
982
983 // Define sub-division width
984 Samples.push_back(dLoE / n_div);
985
986 // Loop over sub-divisions
987 for (int k = 0; k < n_div; k++) {
988 // Define sub-division center
989 double bctr = LoE - dLoE / 2 + (k + 0.5) * dLoE / n_div;

991 Samples.push_back(bctr);
992
993 } // End of loop over sub-divisions
994
995 // Return sample points
996 return Samples;

1049matrixC PMNS_TaylorExp::GetSamplePoints(double InvE, double dInvE, double cosT,
1050 double dcosT)
1051{
1052 // Set a number of sub-divisions to achieve "good" accuracy
1053 // This needs to be studied better
1054 int n_divCosT = ceil(380 * pow(InvE, 0.5) * pow(abs(dcosT / cosT), 0.8) /
1055 sqrt(fAvgProbPrec / 1e-4));
1056 int n_divE = ceil(260 * pow(dInvE / InvE, 0.8) * pow(InvE, 0.6) /
1057 sqrt(fAvgProbPrec / 1e-4));
1058
1059 // A matrix to store sample points
1060 matrixC Samples = matrixC(n_divE + 1, vectorC(n_divCosT + 1, 0));
1061
1062 // Define sub-division width
1063 Samples[0][0] = complexD(dInvE / n_divE, dcosT / n_divCosT);
1064
1065 // Loop over sub-divisions
1066 for (int k = 0; k < n_divE; k++) {
1067 // Define sub-division center for energy
1068 double bctr_InvE = InvE - dInvE / 2 + (k + 0.5) * dInvE / n_divE;
1069
1070 for (int l = 0; l < n_divCosT; l++) {
1071 // Define sub-division center for angle
1072 double bctr_CosT = cosT - dcosT / 2 + (l + 0.5) * dcosT / n_divCosT;
1073
1074 Samples[k + 1][l + 1] = complexD(bctr_InvE, bctr_CosT);
1075 }
1076
1077 } // End of loop over sub-divisions
1078
1079 // Return sample points
1080 return Samples;
1081}

1102double PMNS_TaylorExp::AvgFormulaExtrapolation(int flvi, int flvf, double dbin,
1103 vectorD lambda, matrixC V)
1104{
1105 vectorC SV = vectorC(fNumNus, 0);
1106
1107 for (int i = 0; i < fNumNus; i++) {
1108 for (int k = 0; k < fNumNus; k++) {
1109 SV[i] += fevolutionMatrixS[flvf][k] * V[k][i];
1110 }
1111 }
1112
1113 complexD buffer[3];
1114
1115 for (int n = 0; n < fNumNus; n++) { buffer[n] = SV[n] * conj(V[flvi][n]); }
1116
1117 complexD expo[fNumNus][fNumNus];
1118
1119 for (int j = 0; j < fNumNus; j++) {
1120 for (int i = 0; i < fNumNus; i++) {
1121 double arg = (lambda[j] - lambda[i]) * dbin;
1122 expo[j][i] = exp(complexD(0.0, arg));
1123 }
1124 }
1125
1126 complexD P = 0;
1127
1128 for (int j = 0; j < fNumNus; j++) {
1129 for (int i = 0; i < fNumNus; i++) {
1130 P += buffer[i] * conj(buffer[j]) * expo[j][i];
1131 }
1132 }
1133
1134 return real(P);
1135}
