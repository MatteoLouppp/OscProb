123void PMNS_TaylorExp::BuildKE(double L)
124{
125 double lenghtEV = L * kKm2eV; // L in eV-1
126 double bufK = lenghtEV * 0.5; // L/2 in eV-1
127
128 complexD buffer[3];
129
130 for (int i = 0; i < fNumNus; i++) {
131 complexD Hms_kl;
132
133 for (int l = 0; l < fNumNus; l++) {
134 buffer[l] = 0;
135
136 for (int k = 0; k < fNumNus; k++) {
137 if (k <= l)
138 Hms_kl = fHms[k][l];
139 else
140 Hms_kl = conj(fHms[l][k]);
141
142 if (fIsNuBar && k != l) Hms_kl = conj(Hms_kl);
143
144 buffer[l] += conj(fEvec[k][i]) * Hms_kl;
145 }
146 }
147
148 for (int j = 0; j <= i; j++) {
149 fKmass[i][j] = 0;
150
151 for (int l = 0; l < fNumNus; l++) {
152 fKmass[i][j] += buffer[l] * fEvec[l][j];
153 }
154
155 complexD C;
156
157 if (i == j) { C = complexD(1, 0); }
158 else {
159 double arg = (fEval[i] - fEval[j]) * lenghtEV;
160
161 C = (complexD(cos(arg), sin(arg)) - complexD(1, 0.0)) /
162 complexD(0.0, arg);
163 }
164
165 fKmass[i][j] *= bufK * C;
166
167 if (i != j) fKmass[j][i] = conj(fKmass[i][j]);
168 }
169 }
170}

196//.............................................................................
200double PMNS_TaylorExp::LnDerivative()
201{
202 double dL = 0;
203
204 double L1 = pow(fprem.GetPremLayers()[fLayer].radius, 2) - fminRsq;
205
206 double L2 = -fminRsq;
207 if (fLayer > 0) L2 += pow(fprem.GetPremLayers()[fLayer - 1].radius, 2);
208
209
210 bool ismin = (L2 <= 0 && fcosT < 0);
211
212 if (ismin)
213 dL = 2 * pow(fDetRadius, 2) * fcosT * pow(L1, -0.5);
214 else
215 dL = pow(fDetRadius, 2) * fcosT * (pow(L1, -0.5) - pow(L2, -0.5));
216
217 if (ismin) fdl = 1;
218
219 fLayer += fdl;
220
221 return dL;
222}

350//.............................................................................
355void PMNS_TaylorExp::PropagatePathTaylor(NuPath p)
356{
357 // Set the neutrino path
358 SetCurPath(p);
359
360 // Solve for eigensystem
361 SolveHam();
362
363 // Get the evolution matrix in mass basis
364 double LengthIneV = kKm2eV * p.length;
365 for (int i = 0; i < fNumNus; i++) {
366 double arg = fEval[i] * LengthIneV;
367 fPhases[i] = complexD(cos(arg), -sin(arg));
368 }
369
370 // Rotate S in flavor basis
371 rotateS();
372
373 // if avg on E
374 if (fdInvE != 0) {
375 // Build KE in mass basis
376 BuildKE(p.length);
377
378 // Rotate KE in flavor basis
379 rotateK();
380
381 // Multiply this layer K's with the previous path K's
382 MultiplicationRuleK(fKInvE);
383 }
384
385 // if avg on cosT
386 if (fdcosT != 0) {
387 // Build KcosT in mass basis
388 BuildKcosT(p.length);
389
390 // Multiply this layer K's with the previous path K's
391 MultiplicationRuleK(fKcosT);
392 }
393
394 // Multiply this layer S's with the previous path S's
395 MultiplicationRuleS();
396}

411{

444double PMNS_TaylorExp::AvgFormula(int flvi, int flvf, double dbin,
445 vectorD lambda, matrixC V)
446{
447 vectorC SV = vectorC(fNumNus, 0);
448
449 for (int i = 0; i < fNumNus; i++) {
450 for (int k = 0; k < fNumNus; k++) {
451 SV[i] += fevolutionMatrixS[flvf][k] * V[k][i];
452 }
453 }
454
455 complexD buffer[3];
456
457 for (int n = 0; n < fNumNus; n++) { buffer[n] = SV[n] * conj(V[flvi][n]); }
458
459 complexD sinc[fNumNus][fNumNus];
460
461 for (int j = 0; j < fNumNus; j++) {
462 sinc[j][j] = 1;
463
464 for (int i = 0; i < j; i++) {
465 double arg = (lambda[i] - lambda[j]) * dbin / 2;
466 sinc[i][j] = sin(arg) / arg;
467 sinc[j][i] = sinc[i][j];
468 }
469 }
470
471 complexD P = 0;
472
473 for (int j = 0; j < fNumNus; j++) {
474 for (int i = 0; i < fNumNus; i++) {
475 P += buffer[i] * conj(buffer[j]) * sinc[j][i];
476 }
477 }
478
479 return real(P);
480}

541double PMNS_TaylorExp::AvgProbLoE(int flvi, int flvf, double LoE, double dLoE)
542{
543 if (LoE <= 0) return 0;
544
545 if (fNuPaths.empty()) return 0;
546
547 // Don't average zero width

552
553 double avgprob = 0;

555 double sumw = 0;
556
557 // Loop over all sample points

561 avgprob += w * AvgAlgo(flvi, flvf, samples[j], samples[0], L);
562
563 sumw += w;
564 }
565
566 // Return average of probabilities
567 return avgprob / sumw;

582double PMNS_TaylorExp::AvgAlgo(int flvi, int flvf, double LoE, double dLoE,
583 double L)
584{
585 // Set width bin as 1/E
586 double d1oE = dLoE / L;
587
588 // reset K et S et Ve et lambdaE
589 InitializeTaylorsVectors();
590
591 SetEnergy(L / LoE);
592 SetwidthBin(d1oE, 0);
593
594 // Propagate -> get S and K matrix (on the whole path)
595 PropagateTaylor();
596
597 // DiagolK -> get VE and lambdaE
598 SolveK(fKInvE, flambdaInvE, fVInvE);
599
600 // return fct avr proba
601 return AvgFormula(flvi, flvf, d1oE / kGeV2eV, flambdaInvE, fVInvE);

761double PMNS_TaylorExp::AvgProbLoE(int flvi, int flvf, double LoE, double dLoE,
762 double cosT, double dcosT)
763{
764 if (LoE <= 0) return 0;
765 if (cosT > 0) return 0;

792 imag(samples[k][l]), imag(samples[0][0]));
793
794 sumw += w;
795 }
796 }
797

799 return avgprob / ((sumw));

850double PMNS_TaylorExp::AlgorithmDensityMatrix(int flvi, int flvf)
851{
852 fdensityMatrix[flvi][flvi] = 1;
853
854 RotateDensityM(true, fVcosT);
855 HadamardProduct(flambdaCosT, fdcosT);
856 RotateDensityM(false, fVcosT);
857
858 RotateDensityM(true, fVInvE);
859 HadamardProduct(flambdaInvE, fdInvE / kGeV2eV);
860 RotateDensityM(false, fVInvE);
861
862 RotateDensityM(false, fevolutionMatrixS);
863
864 return real(fdensityMatrix[flvf][flvf]);

875void PMNS_TaylorExp::RotateDensityM(bool to_basis, matrixC V)
876{
877 matrixC Buffer = matrixC(fNumNus, vectorC(fNumNus, 0));
878
879 for (int i = 0; i < fNumNus; i++) {
880 for (int j = 0; j < fNumNus; j++) {
881 for (int k = 0; k < fNumNus; k++) {
882 if (to_basis)
883 Buffer[i][j] += fdensityMatrix[i][k] * V[k][j];
884 else
885 Buffer[i][j] += fdensityMatrix[i][k] * conj(V[j][k]);
886 }
887 }
888 }
889
890 for (int i = 0; i < fNumNus; i++) {
891 for (int j = i; j < fNumNus; j++) {
892 fdensityMatrix[i][j] = 0;
893 for (int k = 0; k < fNumNus; k++) {
894 if (to_basis)
895 fdensityMatrix[i][j] += conj(V[k][i]) * Buffer[k][j];
896 else
897 fdensityMatrix[i][j] += V[i][k] * Buffer[k][j];
898 }
899 if (j > i) fdensityMatrix[j][i] = conj(fdensityMatrix[i][j]);
900 }
901 }

911void PMNS_TaylorExp::HadamardProduct(vectorD lambda, double dbin)
912{
913 matrixC sinc = matrixC(fNumNus, vectorC(fNumNus, 0));
914 for (int j = 0; j < fNumNus; j++) {
915 for (int i = 0; i < j; i++) {
916 double arg = (lambda[i] - lambda[j]) * dbin;
917 sinc[i][j] = sin(arg) / arg;
918
919 sinc[j][i] = sinc[i][j];
920 }
921 }
922
923 for (int i = 0; i < fNumNus; i++) { sinc[i][i] = 1; }
924
925 for (int j = 0; j < fNumNus; j++) {
926 for (int i = 0; i < fNumNus; i++) {
927 fdensityMatrix[i][j] = fdensityMatrix[i][j] * sinc[i][j];
928 }
929 }

942vectorD PMNS_TaylorExp::GetSamplePoints(double LoE, double dLoE)
943{
944 // Set a number of sub-divisions to achieve "good" accuracy
945 // This needs to be studied better
946 int n_div = ceil(3 * pow(dLoE / LoE, 0.8) * pow(LoE, 0.3) /
947 sqrt(fAvgProbPrec / 1e-4));
948
949 // A vector to store sample points
950 vectorD Samples;
951
952 // Define sub-division width
953 Samples.push_back(dLoE / n_div);
954
955 // Loop over sub-divisions
956 for (int k = 0; k < n_div; k++) {
957 // Define sub-division center
958 double bctr = LoE - dLoE / 2 + (k + 0.5) * dLoE / n_div;
959
960 Samples.push_back(bctr);
961
962 } // End of loop over sub-divisions
963
964 // Return sample points
965 return Samples;
966}

1005//.............................................................................
1018matrixC PMNS_TaylorExp::GetSamplePoints(double InvE, double dInvE, double cosT,
1019 double dcosT)
1020{
1021 // Set a number of sub-divisions to achieve "good" accuracy
1022 // This needs to be studied better
1023 int n_divCosT = ceil(380 * pow(InvE, 0.5) * pow(abs(dcosT / cosT), 0.8) /
1024 sqrt(fAvgProbPrec / 1e-4));
1025 int n_divE = ceil(260 * pow(dInvE / InvE, 0.8) * pow(InvE, 0.6) /
1026 sqrt(fAvgProbPrec / 1e-4));
1027
1028 // A matrix to store sample points
1029 matrixC Samples = matrixC(n_divE + 1, vectorC(n_divCosT + 1, 0));
1030
1031 // Define sub-division width
1032 Samples[0][0] = complexD(dInvE / n_divE, dcosT / n_divCosT);
1033
1034 // Loop over sub-divisions
1035 for (int k = 0; k < n_divE; k++) {
1036 // Define sub-division center for energy

1071double PMNS_TaylorExp::AvgFormulaExtrapolation(int flvi, int flvf, double dbin,
1072 vectorD lambda, matrixC V)
1073{
1074 vectorC SV = vectorC(fNumNus, 0);
1075
1076 for (int i = 0; i < fNumNus; i++) {
1077 for (int k = 0; k < fNumNus; k++) {
1078 SV[i] += fevolutionMatrixS[flvf][k] * V[k][i];
1079 }
1080 }
1081
1082 complexD buffer[3];
1083
1084 for (int n = 0; n < fNumNus; n++) { buffer[n] = SV[n] * conj(V[flvi][n]); }
1085
1086 complexD expo[fNumNus][fNumNus];
1087
1088 for (int j = 0; j < fNumNus; j++) {
1089 for (int i = 0; i < fNumNus; i++) {
1090 double arg = (lambda[j] - lambda[i]) * dbin;
1091 expo[j][i] = exp(complexD(0.0, arg));
1092 }
1093 }
1094
1095 complexD P = 0;
1096
1097 for (int j = 0; j < fNumNus; j++) {
1098 for (int i = 0; i < fNumNus; i++) {
1099 P += buffer[i] * conj(buffer[j]) * expo[j][i];
1100 }
1101 }
1102
1103 return real(P);
1104}
