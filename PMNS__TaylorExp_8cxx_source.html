125void PMNS_TaylorExp::BuildKE(double L , matrixC& K)
126{
127 double lenghtEV = L * kKm2eV ; // L in eV-1
128 double bufK = lenghtEV * 0.5 ; // L/2 in eV-1
129
130 complexD buffer[3];
131
132 for(int i = 0 ; i<fNumNus ; i++){
133
134 //complexD buffer[3];
135
136 complexD Hms_kl;
137
138 for(int l = 0 ; l<fNumNus ; l++){
139
140 buffer[l] = 0;
141
142 for(int k = 0 ; k<fNumNus ; k++){
143
144 if (k<=l)
145 Hms_kl = fHms[k][l];
146 else
147 Hms_kl = conj(fHms[l][k]);
148
149 if(fIsNuBar && k!=l)
150 Hms_kl = conj(Hms_kl);
151
152 buffer[l] += conj(fEvec[k][i]) * Hms_kl;
153 }
154 }
155
156 for(int j = 0 ; j<=i ; j++){
157
158 K[i][j] = 0;
159
160 for (int l = 0 ; l<fNumNus ; l++){
161 K[i][j] += buffer[l] * fEvec[l][j] ;
162 }
163
164 complexD C;
165
166 if(i == j){
167 C = complexD(1,0);
168 }
169 else {
170 double arg = (fEval[i] - fEval[j]) * lenghtEV ;
171
172 //C = ( exp(complexD(0.0 , arg)) - complexD(1 , 0.0) ) / complexD(0.0 , arg) ;
173 C = ( complexD(cos(arg) , sin(arg)) - complexD(1 , 0.0) ) / complexD(0.0 , arg) ;
174
175 }
176
177 K[i][j] *= bufK * C ;
178
179 if(i != j)
180 K[j][i] = conj(K[i][j]);
181
182 }
183 }

194void PMNS_TaylorExp::BuildKcosT(double L, matrixC& K)
195{
196 double lv = 2 * kGeV2eV * fEnergy; // 2E in eV
197 double kr2GNe = kK2 * M_SQRT2 * kGf;
198 kr2GNe *= fPath.density * fPath.zoa; // Matter potential in eV
199
200 matrixC Hbar = matrixC(fNumNus, vectorC(fNumNus, 0));
201
202 for (int i = 0; i < fNumNus; i++) {
203 Hbar[i][i] = fHms[i][i] / lv;
204 for (int j = i + 1; j < fNumNus; j++) {
205 if (!fIsNuBar)
206 Hbar[i][j] = fHms[i][j] / lv;
207 else
208 Hbar[i][j] = conj(fHms[i][j]) / lv;
209 }
210 }
211 if (!fIsNuBar)
212 Hbar[0][0] += kr2GNe;
213 else
214 Hbar[0][0] -= kr2GNe;
215
216 double dL = LnDerivative() * kKm2eV;
217
218 for(int j = 0 ; j<fNumNus ; j++){
219 for(int i = 0 ; i<=j ; i++){
220 K[i][j] = dL * Hbar[i][j];
221
222 if(i != j){
223 K[j][i] = conj(K[i][j]);
224 }
225 }
226 }
227

234double PMNS_TaylorExp::LnDerivative()
235{
236 double dL = 0;
237
238 double L1 = pow(fPremLayers[flayer].radius, 2) - fminRsq;
239
240 double L2 = -fminRsq;
241 if (flayer > 0) L2 += pow(fPremLayers[flayer - 1].radius, 2);
242
243 bool ismin = (L2 <= 0 && fcosT < 0);
244
245 if (ismin)
246 dL = 2 * pow(fDetRadius,2) * fcosT * pow( L1 , -0.5) ;
247 else
248 dL = pow(fDetRadius,2) * fcosT * ( pow( L1 , -0.5) - pow( L2 , -0.5));
249
250 if (ismin) fdl = 1;
251
252 flayer += fdl;
253
254 return dL;
255}

403void PMNS_TaylorExp::PropagatePathTaylor(NuPath p)
404{
405
406 // Set the neutrino path
407 SetCurPath(p);
408
409 // Solve for eigensystem
410 SolveHam();
411
412 // Get the evolution matrix in mass basis
413 double LengthIneV = kKm2eV * p.length;
414 for (int i = 0; i < fNumNus; i++) {
415 double arg = fEval[i] * LengthIneV;
416 fPhases[i] = complexD(cos(arg), -sin(arg));
417 }
418
419 // Rotate S in flavor basis
420 //matrixC fSflavor = matrixC(fNumNus, vectorC(fNumNus, 0));
421 rotateS();
422
423 // if avg on E
424 if(fdInvE != 0){
425
426 // Build KE in mass basis
427 //matrixC fKmass = matrixC(fNumNus, vectorC(fNumNus, 0));
428 BuildKE(p.length,fKmass);
429
430 // Rotate KE in flavor basis
431 //matrixC fKflavor = matrixC(fNumNus, vectorC(fNumNus, 0));
432 rotateK(fKmass,fKflavor);
433
434 // Multiply this layer K's with the previous path K's
435 MultiplicationRuleK(fKflavor,fKInvE);
436
437 }
438
439 // if avg on cosT
440 if(fdcosT != 0){
441
442 //matrixC fKmass2 = matrixC(fNumNus, vectorC(fNumNus, 0));
443 BuildKcosT(p.length, fKflavor);
444
445 // Multiply this layer K's with the previous path K's
446 MultiplicationRuleK(fKflavor,fKcosT);
447
448 }
449
450
451
452
453 // Multiply this layer S's with the previous path S's
454 MultiplicationRuleS(fSflavor);
455
456}

504double PMNS_TaylorExp::AvgFormula(int flvi, int flvf, double dbin, vectorD lambda, matrixC V)
505{
506
507 vectorC SV = vectorC(fNumNus, 0);
508
509 for(int i = 0 ; i<fNumNus ; i++) {
510 for(int k = 0 ; k<fNumNus ; k++){
511 SV[i] += fevolutionMatrixS[flvf][k] *V[k][i];
512 }
513 }
514
515 complexD buffer [3];
516
517 for( int n = 0 ; n<fNumNus ; n++){
518 buffer[n] = SV[n] * conj(V[flvi][n]);
519 }
520
521 complexD sinc[fNumNus][fNumNus];
522
523 for(int j = 0 ; j<fNumNus ; j++){
524
525 sinc[j][j] = 1;
526
527 for(int i = 0 ; i<j ; i++){
528 double arg = (lambda[i] - lambda[j]) * dbin / 2;
529 sinc[i][j] = sin(arg) / arg;
530 sinc[j][i] = sinc[i][j];
531 }
532 }
533
534 complexD P = 0;
535
536 for(int j = 0 ; j<fNumNus ; j++){
537 for(int i = 0 ;i<fNumNus ;i++){
538 P += buffer[i] *conj(buffer[j]) * sinc[j][i];
539 }
540 }
541
542 return real(P);
543}

567{

599double PMNS_TaylorExp::AvgProbLoE(int flvi, int flvf, double LoE , double dLoE)
600{
601 if (LoE <= 0) return 0;
602
603 if (fNuPaths.empty()) return 0;
604
605 // Don't average zero width
606 if (dLoE <= 0) return Prob(flvi, flvf, fPath.length / LoE);
607
608 // Get sample points for this bin
609 vectorD samples = GetSamplePoints(LoE, dLoE);
610
611 double avgprob = 0;
612 double L = fPath.length;
613 double sumw = 0;
614
615 // Loop over all sample points
616 for (int j = 1; j < int(samples.size()); j++) {
617
618 double w = 1. / pow(samples[j], 2);
619
620 avgprob += w * AvgAlgo(flvi, flvf, samples[j], samples[0],L);
621
622 sumw += w;
623
624 }
625
626 // Return average of probabilities
627 return avgprob / sumw;
628}

900{
901
902 // Set a number of sub-divisions to achieve "good" accuracy
903 // This needs to be studied better

905 //int n_div = ceil(10* pow(dLoE / LoE, 0.8) * pow(LoE, 0.2) / sqrt(fAvgProbPrec / 1e-4));

907
908 // A vector to store sample points
909 vectorD Samples;
910 Samples.push_back(dLoE / n_div);
911
912 // Loop over sub-divisions
913 for (int k = 0; k < n_div; k++) {
914 // Define sub-division center and width
915 double bctr = LoE - dLoE / 2 + (k + 0.5) * dLoE / n_div;
916
917 Samples.push_back(bctr);
918
919 } // End of loop over sub-divisions
920
921 // Return all sample points
922 return Samples;
923}

959void PMNS_TaylorExp::RotateDensityM(bool to_mass, matrixC V, matrixC& densityMatrix)
960{
961 matrixC Buffer = matrixC(fNumNus, vectorC(fNumNus, 0));
962
963 // buffer = rho . U
964 for (int i = 0; i < fNumNus; i++) {
965 for (int j = 0; j < fNumNus; j++) {
966 for (int k = 0; k < fNumNus; k++) {
967 if (to_mass)
968 Buffer[i][j] += densityMatrix[i][k] * V[k][j];
969 else
970 Buffer[i][j] += densityMatrix[i][k] * conj(V[j][k]);
971 }
972 }
973 }
974
975 // rho = U^\dagger . buffer = U^\dagger . rho . U
976 // Final matrix is Hermitian, so copy upper to lower triangle
977 for (int i = 0; i < fNumNus; i++) {
978 for (int j = i; j < fNumNus; j++) {
979 densityMatrix[i][j] = 0;
